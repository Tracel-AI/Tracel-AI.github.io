pub enum Initializer {
    Constant {
        value: f64,
    },
    Ones,
    Zeros,
    Uniform {
        min: f64,
        max: f64,
    },
    Normal {
        mean: f64,
        std: f64,
    },
    KaimingUniform {
        gain: f64,
        fan_out_only: bool,
    },
    KaimingNormal {
        gain: f64,
        fan_out_only: bool,
    },
    XavierUniform {
        gain: f64,
    },
    XavierNormal {
        gain: f64,
    },
}

Enum specifying with what values a tensor should be initialized
